5. Public Capabilities:
	Product: 
		"give me id, name and price, I will create a product"
		- Represent immutable product data (id, name, price)
	Cart: 
		"give me product and quantity, I will add it into the cart"
		- Accept a product and quantity to add
		"give me product, I will remove it from cart with quantity"
		- Remove an existing product
		"give me quantity of existing product, I will update the quantity"
		- Update quantity of an existing product
		"call me, I will calculate the total price of cart items"
		- Provide cart subtotal
		- Provide list of cart items
	Discount: 
		"give me subtotal amount, I will tell you discount amount"
		- Given an amount, return discount amount
	Tax: 
		"give me discounted amount, I will tell you the tax amount"
		- Given an amount, return tax amount
		"give me tax rate, I will reconfigure it"
		- Allow tax rate to be configured
	Billing: 
		"call me, I will generate bill"
		- Use cart data to compute billing summary
		- Coordinate discount and tax calculation
		- Coordinate discount and tax calculation
	Invoice: 
		"call me, I will print invoice in console"
		- Accept billing result
		- Format invoice output
		- Print invoice to console
	Storage: 
		"Call me, I will store bill in file/db/json"
		- Accept invoice data
		- Persist invoice to file (overwrite)
	UserInteraction: 
		"I will display menu, read user input"
		- Display menu
		- Read user input
		- Trigger cart and billing actions

9. Class Design:

class Product:
	- id: int
	- name: string
	- price: double
	+ Product(id: int, name: string, price: double);
	+ const getId(): int
	+ const getName(): string
	+ const getPrice(): double

class CartItem:
	- product: const Product*
	- quantity: int
	+ CartItem(product: const Product*, quantity: int);
	+ const getProduct(): const Product*
	+ const getQuantity(): int

class Cart:
	- cartItems: vector<CartItem>
	+ addProduct(product: const Product*, quantity: int): void
	+ removeProduct(product: const Product*): void
	+ updateQuantity(product: const Product*, quantity: int): void
	+ const calculateSubTotal(): double
	+ const getItems(): vector<CartItem>&

class Discount:
	+ const calculateDiscount(subtotal: double) as pure virtual: double
	+ ~Discount(): default

class FlatDiscount : + Discount:
	- flatAmount: double
	+ FlatDiscount(flatAmount: double);

class PercentageDiscount : + Discount:
	- percentage: double
	+ PercentageDiscount(percentage: double);

class ThresholdDiscount : + Discount:
	- thresholdAmount: double
	+ ThresholdDiscount(thresholdAmount: double, discountAmount: double);

class Tax:
	- taxRate: double
	+ Tax(taxRate: double);
	+ const calcualateTax(amountAfterDiscount: double): double
	+ setTaxRate(taxRate: double): void

class BillData:
	+ subtotal: double
	+ totalDiscount: double
	+ taxAmount: double
	+ finalAmount: double
	+ items: vector<CartItem>

class Billing:
	+ const generateBill(cart: const Cart&, discounts: const vector<Discount*>, tax: const Tax&): BillData

class Invoice:
	+ const printToConsole(billData: const BillData&): void
	+ const generateInvoiceText(billData: const BillData&): string

class Storage:
	+ const save(invoiceText: const stringa&): void
	+ ~Storage(): default
